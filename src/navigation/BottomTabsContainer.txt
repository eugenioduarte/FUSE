import { ROUTES } from '@/constants/routes'
import { useTheme } from '@/hooks/useTheme'
import { navigatorManager } from '@/navigation/navigatorManager'
import { useBottomTabStore } from '@/store/useBottomTabStore'
import { useOverlay } from '@/store/useOverlay'
import { Ionicons } from '@expo/vector-icons'
import React from 'react'
import { StyleSheet, TouchableOpacity, View } from 'react-native'

import CalendarScreen from '@/screens/CalendarScreen/CalendarScreen'
import DashboardScreen from '@/screens/main/dashboard/DashboardScreen'
import TopicScreen from '@/screens/main/topic/TopicScreen'
import { createStackNavigator } from '@react-navigation/stack'

const Stack = createStackNavigator()

export default function BottomTabsContainer() {
  const theme = useTheme()
  const setFastWayOverlay = useOverlay((s) => s.setFastWayOverlay)
  const { activeTab, setActiveTab } = useBottomTabStore()

  const handleNavigate = (screen: 'dashboard' | 'topics' | 'calendar') => {
    setActiveTab(screen)
    switch (screen) {
      case 'topics':
        navigatorManager.goToTopic()
        break
      case 'calendar':
        navigatorManager.goToCalendar()
        break
      default:
        navigatorManager.goToDashboard()
        break
    }
  }

  const handleFastWayPress = () => {
    setFastWayOverlay(true)
  }

  return (
    <View style={styles.container}>
      <View style={styles.stackContainer}>
        <Stack.Navigator screenOptions={{ headerShown: false }}>
          <Stack.Screen
            name={ROUTES.DashboardScreen}
            component={DashboardScreen}
          />
          <Stack.Screen name={ROUTES.TopicScreen} component={TopicScreen} />
          <Stack.Screen
            name={ROUTES.CalendarScreen}
            component={CalendarScreen}
          />
        </Stack.Navigator>
      </View>

      <View
        style={[
          styles.bottomBar,
          { backgroundColor: theme.colors.backgroundPrimary },
        ]}
      >
        <TouchableOpacity
          style={styles.tab}
          onPress={() => handleNavigate('dashboard')}
        >
          <Ionicons
            name="home-outline"
            size={26}
            color={
              activeTab === 'dashboard'
                ? theme.colors.accentRed
                : theme.colors.textSecondary
            }
          />
        </TouchableOpacity>

        <TouchableOpacity
          style={styles.tab}
          onPress={() => handleNavigate('topics')}
        >
          <Ionicons
            name="book-outline"
            size={26}
            color={
              activeTab === 'topics'
                ? theme.colors.accentRed
                : theme.colors.textSecondary
            }
          />
        </TouchableOpacity>

        <TouchableOpacity
          style={styles.tab}
          onPress={() => handleNavigate('calendar')}
        >
          <Ionicons
            name="calendar-outline"
            size={26}
            color={
              activeTab === 'calendar'
                ? theme.colors.accentRed
                : theme.colors.textSecondary
            }
          />
        </TouchableOpacity>

        <TouchableOpacity style={styles.tab} onPress={handleFastWayPress}>
          <Ionicons
            name="flash-outline"
            size={26}
            color={theme.colors.accentOrange}
          />
        </TouchableOpacity>
      </View>
    </View>
  )
}

const styles = StyleSheet.create({
  container: { flex: 1 },
  stackContainer: { flex: 1 },
  bottomBar: {
    flexDirection: 'row',
    justifyContent: 'space-around',
    alignItems: 'center',
    borderTopWidth: 2,
    borderColor: '#3A001D',
    paddingVertical: 12,
  },
  tab: {
    flex: 1,
    alignItems: 'center',
  },
})
